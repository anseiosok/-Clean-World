    function setField(field, value) {
      state[field] = value;

      ['sol', 'nublado', 'chuva_intensa', 'nenhuma', 'leve', 'severa', 'nao', 'sim', 'verde', 'piscando', 'vermelho'].forEach(opt => {

        if (btn) {

            btn.className = "py-2 px-1 rounded-lg border text-center transition-all border-slate-800 bg-slate-900/40 text-slate-400";
          }
        }
      });

      calculateDiagnostic();
    }

    function updateSelect(field, val) {
      state[field] = val;
      calculateDiagnostic();
    }

    // Algoritmo de Diagnóstico de Desempenho
    function calculateDiagnostic() {
      let score = 100;
      let alerts = [];
      let recommendations = [];
      let needProfessional = false;

      if (state.clima === 'nublado') {
        alerts.push("Produção reduzida (10% a 25%) devido à nebulosidade.");
        recommendations.push("Os painéis captam luz difusa em dias cinzas. É perfeitamente normal.");
      } else if (state.clima === 'chuva_intensa') {
        alerts.push("Geração reduzida de forma temporária pela chuva.");
        recommendations.push("A chuva forte auxilia na limpeza prévia da poeira leve nas placas.");
      }

      if (state.sujeira === 'leve') {
        score -= 10;
        alerts.push("Resíduos de poeira leve acumulados.");
        recommendations.push("Se possível, programe uma lavagem simples apenas com água limpa.");
      } else if (state.sujeira === 'severa') {
        score -= 25;
        alerts.push("Sujeira severa (fezes secas, fuligem). Perda de eficiência.");
        if (state.regiao === 'rural_praia' || state.ultimoServico === 'mais6meses') {
          needProfessional = true;
          recommendations.push("Recomendado suporte C&M Solar devido à dureza dos resíduos.");
        } else {
          recommendations.push("Realize uma lavagem criteriosa ou chame auxílio técnico.");
        }
      }

      if (state.sombreamento === 'sim') {
        score -= 30;
        alerts.push("Sombras frequentes detectadas sobre os módulos.");
        recommendations.push("Avalie poda de copas de árvores ou reposicionamento de obstáculos.");
      }

      if (state.inversor === 'vermelho' || state.inversor === 'piscando') {
        score -= 50;
        alerts.push("Erro elétrico indicado no visor do Inversor.");
        needProfessional = true;
        recommendations.push("Não toque nos aparelhos. Chame o suporte técnico imediatamente.");
      }

      if (state.ultimoServico === 'mais6meses') {
        if (state.regiao === 'rural_praia') {
          alerts.push("Excedido período seguro para maresia/zona rural (6 meses).");
          needProfessional = true;
          recommendations.push("Manutenções preventivas semestrais em Sergipe evitam corrosão.");
        } else {
          alerts.push("Usina urbana necessitando de revisão anual.");
          recommendations.push("Considere agendar limpeza ou fazer autovistoria visual.");
        }
      }

      let status = "Saudável";
      let textClass = "text-emerald-400 border-emerald-500/30 bg-emerald-500/10";

      if (score < 55 || needProfessional) {
        status = "Visita Técnica Indicada";
        textClass = "text-rose-400 border-rose-500/30 bg-rose-500/10";
      } else if (score < 85) {
        status = "Atenção Necessária";
        textClass = "text-amber-400 border-amber-500/30 bg-amber-500/10";
      }

      document.getElementById('status-badge').innerText = status;
      document.getElementById('status-badge').className = `inline-block border px-2 py-0.5 rounded-full text-[8px] font-black uppercase tracking-wider ${textClass}`;
      document.getElementById('gauge-value').innerText = `${score}%`;

      const circle = document.getElementById('gauge-circle');
      const circumference = 213.62;
      const offset = circumference * (1 - score / 100);
      circle.style.strokeDashoffset = offset;

      const alertsContainer = document.getElementById('alerts-container');
      if (alerts.length === 0) {
        alertsContainer.innerHTML = `
          <div class="text-[10px] text-emerald-400 bg-emerald-950/20 p-2 rounded-xl border border-emerald-900/30 flex items-center gap-1.5">
            <span>✓</span> Usina operando de forma otimizada.
          </div>`;
      } else {
        alertsContainer.innerHTML = alerts.map(al => `
          <div class="text-[10px] text-amber-300 bg-amber-950/20 p-2 rounded-xl border border-amber-900/30 flex items-start gap-1.5">
            <span class="mt-0.5">⚠️</span> <span>${al}</span>
          </div>`).join('');
      }

      const recsContainer = document.getElementById('recommendations-container');
      const standardRecs = recommendations.length > 0 ? recommendations : ["Nenhum procedimento imediato necessário. Continue acompanhando os indicadores semanais."];
      recsContainer.innerHTML = standardRecs.map(rec => `
        <div class="text-[10px] text-emerald-300 bg-emerald-950/15 p-2 rounded-xl border border-emerald-900/30 flex items-start gap-1.5">
          <span class="mt-0.5">💡</span> <span>${rec}</span>
        </div>`).join('');

      document.getElementById('support-banner').classList.toggle('hidden', !needProfessional);
    }

    // Gerenciador de Checklists Mobile
    function toggleCheckItem(key) {
      state.checklists[key] = !state.checklists[key];

      const chk = document.getElementById(`chk-${key}`);
      if (state.checklists[key]) {
        chk.className = "w-4 h-4 rounded border shrink-0 flex items-center justify-center bg-amber-500 border-amber-500 text-slate-950 text-[10px] font-black";
        chk.innerText = "✓";
      } else {
        chk.className = "w-4 h-4 rounded border border-slate-700 shrink-0 text-transparent flex items-center justify-center";
        chk.innerText = "✓";
      }

      // Calcula progresso total
      const checkedKeys = Object.keys(state.checklists);
      const total = checkedKeys.length;
      const checkedCount = checkedKeys.filter(k => state.checklists[k]).length;
      const percent = Math.round((checkedCount / total) * 100);

      document.getElementById('checklist-progress-bar').style.width = `${percent}%`;
      document.getElementById('checklist-progress-text').innerText = `${percent}%`;

      const hasSpecialCase = state.checklists.manchasResistentes || state.checklists.telhadoInclinado;
      document.getElementById('triagem-alert-banner').classList.toggle('hidden', !hasSpecialCase);
    }

    // Inicialização
